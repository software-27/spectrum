import Button from '@react/react-spectrum/Button';
import CoachMark from '@react/react-spectrum/CoachMark';
import Tour from '@react/react-spectrum/Tour';

## Overview

Tours are used to display an ordered series of CoachMarks on a page. It tracks
how far along the user is in that series and provides hooks to know when the tour
is completed or skipped.

## Example

export class DelayTour extends React.Component {
  constructor() {
    super();
    this.state = { show: false };
  }
  onClick() {
    this.setState({ show: true });
  }
  render() {
    return (!this.state.show ? <Button onClick={this.onClick.bind(this)}>Start Tour</Button> : (
      <Tour disableProgress>
        <CoachMark title="Step 1" selector="#stepOne" confirmLabel="Next" cancelLabel="Skip">
          This is step 1
        </CoachMark>
        <CoachMark title="Step 2" selector="#stepTwo" confirmLabel="Next" cancelLabel="Skip">
          This is step 2
        </CoachMark>
        <CoachMark title="Step 3" selector="#stepThree" confirmLabel="Done">
          You did it!
        </CoachMark>
      </Tour>
    ))
  }
}

<div style={{position: 'relative'}}>
  <div
    style={{
      display: 'flex',
      justifyContent: 'space-between'
    }}>
    <Button id="stepOne">Step One</Button>
    <Button id="stepTwo">Step Two</Button>
    <Button id="stepThree">Step Three</Button>
    <DelayTour />
  </div>
</div>

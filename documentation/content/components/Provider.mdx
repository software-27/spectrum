import Provider from '@react/react-spectrum/Provider';
import Button from '@react/react-spectrum/Button';

## Overview

Provider wraps child components with div and applies css class spectrum--_theme_ and spectrum--_scale_
to the div component.  Both theme and scale can be used together.  Provider can also be nested if needed.

## Example

### Darkest theme

<Provider theme="darkest" scale="large">
  <Button variant="primary">Hello world</Button>
</Provider>

```jsx
<Provider theme="darkest" scale="large">
  <Button variant="primary">Hello world</Button>
</Provider>
```

### Dark theme

<Provider theme="dark" scale="large">
  <Button variant="primary">Hello world</Button>
</Provider>

```jsx
<Provider theme="dark" scale="large">
```

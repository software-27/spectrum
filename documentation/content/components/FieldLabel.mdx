import FieldLabel from '@react/react-spectrum/FieldLabel';
import Textfield from '@react/react-spectrum/Textfield';

## Overview

A **FieldLabel** displays a combined element of Textfield and Label.

## Examples

### Standard Field label

<FieldLabel label="React" labelFor="foo">
  <Textfield placeholder="React" id="foo" />
</FieldLabel>

```jsx
<FieldLabel label="React" labelFor="foo">
  <Textfield placeholder="React" id="foo" />
</FieldLabel>
```


### Position of Field Label

<FieldLabel label="React" label="foo" position="left">
  <Textfield placeholder="React" id="foo" />
</FieldLabel>

```jsx
<FieldLabel label="React" label="foo" position="left">
  <Textfield placeholder="React" id="foo" />
</FieldLabel>
```

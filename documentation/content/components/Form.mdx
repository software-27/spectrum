import Checkbox from '@react/react-spectrum/Checkbox';
import {Form, FormItem} from '@react/react-spectrum/Form';
import Link from '@react/react-spectrum/Link';
import Textarea from '@react/react-spectrum/Textarea';
import Textfield from '@react/react-spectrum/Textfield';

## Overview

For building cohesive web forms with table alignment qualities.

## Examples

### Standard Form

<Form>
  <FormItem label="Company Title">
    <Textfield placeholder="Company Title" />
  </FormItem>
  <FormItem label="Life Story">
    <Textarea placeholder="Life Story" />
  </FormItem>
  <FormItem label={<Link href="#">Terms</Link>}>
    <Checkbox label="I Agree" />
  </FormItem>
</Form>
